<body>
  <script>
    function setup() {
      createCanvas(600, 600);
      background(255);
      noLoop();
    }

    function draw() {
      translate(width / 2, height / 2);
     
      // Draw petals
      fill(255, 105, 180); // Pink
      for (let i = 0; i < 10; i++) {
        rotate(PI / 5);
        ellipse(0, -50, 100, 60);
      }
     
      // Draw center
      fill(255, 255, 0); // Yellow
      circle(0, 0, 50);
     
      // Draw stem
      stroke(0, 128, 0);
      strokeWeight(8);
      line(0, 50, 0, 250);
     
      // Draw leaf
      noStroke();
      fill(144, 238, 144);
      ellipse(30, 200, 60, 30);
     
      // Add text
      fill(128, 0, 128);
      textSize(24);
      textAlign(CENTER);
      text("Happy Girlfriend's Day!", 0, -150);
    }
  </script>
